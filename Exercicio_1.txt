CREATE DATABASE "Exercicio_1"
    WITH 
    OWNER = postgres
    ENCODING = 'UTF8'
    LC_COLLATE = 'Portuguese_Brazil.1252'
    LC_CTYPE = 'Portuguese_Brazil.1252'
    TABLESPACE = pg_default

   DROP TABLE IF EXISTS LOCAL;
   
CREATE TABLE ACADEMICO
(
ID_ACADEMICO INT,
RA INT,
NOME VARCHAR(45),
NASCIMENTO    DATE,
CONSTRAINTS ACADEMICO_PKEY PRIMARY KEY (ID_ACADEMICO)
);
CREATE TABLE CATEGORIA
(
IDCATEGORIA INT,
DESCRI큐O VARCHAR(60),
CONSTRAINTS CATEGORIA_PKEY PRIMARY KEY (IDCATEGORIA)   
);

CREATE TABLE LOCAL(
ID_ACADEMICO INT,
IDLOCAL INT,
NOME VARCHAR(50),
ENDERE큞 VARCHAR(60),
NUMERO INT,
COMPLEMENTO VARCHAR(45),
TELEFONE VARCHAR(15),
CELULAR VARCHAR(15),
CONSTRAINTS LOCAL_PKEY PRIMARY KEY (IDLOCAL),
CONSTRAINTS ACADEMICO_FKEY FOREIGN KEY (ID_ACADEMICO)
);
CREATE TABLE MATRICULA(
IDMATRICULA INT,
ID_ACADEMICO INT,
IDCURSO INT,
DATA_EFETIVA큐O DATE,
DATA_CONCLUSAO DATE,
DATA_INTERRUP큐O DATE,
CONSTRAINTS MATRICULA_PKEY PRIMARY KEY (IDMATRICULA),
CONSTRAINTS ACADEMICO_FKEY FOREIGN KEY (ID_ACADEMICO),
CONSTRAINTS CURSO_FKEY FOREIGN KEY (IDCURSO)
);
CREATE TABLE CURSO(
IDCURSO INT,
IDCATEGORIA INT,
DESCRI큐O VARCHAR(45),
DURA큐O INT,
CARGA_HORARIA INT,
PRESENCIAL VARCHAR(1);
MENSALIDADE NUMERIC,
CONSTRAINTS CURSO_PKEY PRIMARY KEY (IDCURSO),
CONSTRAINTS CATEGORIA_FKEY FOREIGN KEY (IDCATEGORIA)
);

CREATE TABLE DISCIPLINA(
IDDISCIPLINA INT,
IDCURSO INT,
COD_DISCIPLINA VARCHAR(100);
DESCRI큐O VARCHAR(100);
CARGA_HORARIA INT,
VALOR NUMERIC,
CONSTRAINTS DISCIPLINA_PKEY PRIMARY KEY (IDDISCIPLINA),
CONSTRAINTS CURSO_FKEY FOREIGN KEY (IDCURSO)

);

CREATE TABLE BOLETIM(
IDBOLETIM INT,
IDMATRICULA INT,
IDDISCIPLINA INT,
NOTA01 NUMERIC,
NOTA02 NUMERIC,
NOTA03 NUMERIC,
NOTA04 NUMERIC,
CONSTRAINTS BOLETIM_PKEY PRIMARY KEY (IDBOLETIM),
CONSTRAINTS MATRICULA_FKEY FOREIGN KEY (IDMATRICULA),
CONSTRAINTS DISCIPLINA_FKEY FOREIGN KEY (IDDISCIPLINA)
);
